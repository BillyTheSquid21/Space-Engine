52 void GenerateQuadArrayIndices(T& quadArr)
53 {
54 //Buffer indices to minimise counts of data sent to render queue
55 //Find total ints needed
56 unsigned int quadIntCount = quadArr.quadCount * Primitive::Q_IND_COUNT;
57
58 //Resize vector to be able to fit and init index
59 quadArr.indices.resize(quadIntCount);
60 unsigned int indicesIndex = 0;
61
62 unsigned int lastLargest = -1;
63 unsigned int indicesTemp[6]{ 0,1,2,0,2,3 };
64 for (unsigned int i = 0; i < quadArr.quadCount; i++) {
65 //Increment all by last largest - set temp to base
66 std::copy(&Primitive::Q_IND[0], &Primitive::Q_IND[Primitive::Q_IND_COUNT], &indicesTemp[0]);
67 for (int j = 0; j < Primitive::Q_IND_COUNT; j++) {
68 indicesTemp[j] += lastLargest + 1;
69 }
70 //Set last largest
71 lastLargest = indicesTemp[Primitive::Q_IND_COUNT - 1];
72 //Copy into vector
73 std::copy(&indicesTemp[0], &indicesTemp[Primitive::Q_IND_COUNT], quadArr.indices.begin() + indicesIndex);
74 //Increment index
75 indicesIndex += Primitive::Q_IND_COUNT;
76 }

165 template<typename T>
166 static void AxialRotateInternal(void* verticesArray, glm::vec3 rotationCentre, float angle, size_t verticeCount, SGRender::Axis axis)
167 {
168 using namespace SGRender;
169 T* vertexPointer = (T*)verticesArray;
170 glm::vec3 axisVector = { 1.0f, 0.0f, 0.0f };
171 switch (axis)
172 {
173 case Axis::X:
174 axisVector = { 1.0f, 0.0f, 0.0f };
175 break;
176 case Axis::Y:
177 axisVector = { 0.0f, 1.0f, 0.0f };
178 break;
179 case Axis::Z:
180 axisVector = { 0.0f, 0.0f, 1.0f };
181 break;
182 }
183
184 //Translate for each vertice
185 for (int i = 0; i < verticeCount; i++) {
186 Vertex* vertex = (Vertex*)(void*)&vertexPointer[i];
187 glm::vec3 position = { vertex->position.x - rotationCentre.x, vertex->position.y - rotationCentre.y, vertex->position.z - rotationCentre.z };
188 position = glm::rotate(position, glm::radians(angle), axisVector);
189 vertex->position = { position.x + rotationCentre.x, position.y + rotationCentre.y, position.z + rotationCentre.z };
190 }
191 }
