118 std::this_thread::sleep_for(milliseconds(sleepDuration.count()));
119 }
120
121 //Check to reset mean of frames
122 if (deltaFrameCountLong == meanPeriodLong) {
123 deltaFrameCountLong = 1;

126

128 deltaFrameCountShort = 1;

130 }
131
132 //get time at end of all processes - time for one whole cycle
133 endTime = system_clock::now();

136 deltaTimeShort = deltaTimeShort + (((totalTime).count() / 1000.0) - deltaTimeShort) / deltaFrameCountShort;
137 deltaTime = (deltaTimeLong + deltaTimeShort) / 2.0f;
138 }
