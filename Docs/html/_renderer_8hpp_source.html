154 int indiceSizeIndex = 0;

158 {
159 //Get instructions from render queue
160 RenderInfo instructions = m_RenderInstructions[i];
161
162 //VERTICES
163 const float* dataPointer = (const float*)instructions.verts;
164 unsigned int dataSize = instructions.vertFloats;
165
166 //Copy vertices into vector
167 std::copy(&dataPointer[0], &dataPointer[dataSize], verticesIterator + vertIndex);
168 vertIndex += dataSize;
169
170 //INDICES
171 const unsigned int* indDataPointer = instructions.inds;
172 unsigned int indDataSize = instructions.indCount;
173
174 //Add to vector one by one
175 unsigned int currentLargest = 0;
176 for (int j = 0; j < indDataSize; j++) {
177 unsigned int newValue = indDataPointer[j] + largestInd;
178 indices[indIndex] = newValue + indiceSizeIndex;
179 indIndex++;
180 if (newValue > currentLargest) {
181 currentLargest = newValue;
182 }
183 }

186 indiceSizeIndex++;
187
188 }

190 m_IB.bufferData(indices.data(), indices.size());

194 {
195 m_RenderInstructions.shrinkTo(m_VerticesIndex);

197 m_VerticesIndex = 0;
