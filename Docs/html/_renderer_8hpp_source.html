97 {
98 //First get amount of data among all vertice floats
99 unsigned int totalVertFloats = m_PrimitiveVertices.vertFloatCount();
100 //Get amount of data among all indice ints
101 unsigned int totalIndFloats = m_PrimitiveVertices.indIntCount();
102
103 //Then create buffer with space for that many floats
104 std::vector<float> vertices; std::vector<unsigned int> indices; //Buffer declaration
105 vertices.resize(totalVertFloats); indices.resize(totalIndFloats); //Buffer resizing
106 auto verticesIterator = vertices.begin(); auto indicesIterator = indices.begin();
107
108 //Indexes
109 unsigned int vertIndex = 0; unsigned int indIndex = 0;
110 int vertexSizeIndex = 0; int indiceSizeIndex = 0;
111 unsigned int largestInd = 0;
112
113 while (verticesArray.itemsWaiting()) {
114 //Get instructions from render queue
115 BulkRenderContainer<T*> instructions = verticesArray.nextInQueue();
116
117 //For each bulk chunk, render all
118 for (int i = 0; i < instructions.elementsCount; i++)
119 {
120 //VERTICES
121 const float* dataPointer = (const float*)instructions.verts[i];
122 unsigned int dataSize = instructions.vertFloats[i];
123 vertexSizeIndex++;
124
125 //Copy vertices into vector
126 std::copy(&dataPointer[0], &dataPointer[dataSize], verticesIterator + vertIndex);
127 vertIndex += dataSize;
128
129 //INDICES
130 const unsigned int* indDataPointer = instructions.inds[i];
131 unsigned int indDataSize = instructions.indCount[i];
132 //Add to vector one by one
133 unsigned int currentLargest = 0;
134 for (int j = 0; j < indDataSize; j++) {
135 unsigned int newValue = indDataPointer[j] + largestInd;
136 indices[indIndex] = newValue + indiceSizeIndex;
137 indIndex++;
138 if (newValue > currentLargest) {
139 currentLargest = newValue;
140 }
141 }
142 largestInd = currentLargest;
143
144 indiceSizeIndex++;
145 }
146
147 }
148 m_VB.bufferData(vertices.data(), vertices.size());
149 m_IB.bufferData(indices.data(), indices.size());
150 }
