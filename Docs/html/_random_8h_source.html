93 void generateField(int data, float noiseReduction)
94 {
95 //Find corners and dot product
96 for (int y = 0; y < m_Size; y++)
97 {
98 for (int x = 0; x < m_Size; x++)
99 {
100 //Get the coordinates in grid space
101 glm::vec2 pos;
102 pos.x = ((float)x / (float)m_Size) * (float)m_Resolution;
103 pos.y = ((float)y / (float)m_Size) * (float)m_Resolution;
104
105 //Displace slightly
106 pos.x += random.next() / noiseReduction;
107 pos.y += random.next() / noiseReduction;
108
109 //Get bottom left corner
110 glm::vec2 corners[4];
111 corners[0].x = floor(pos.x); corners[0].y = floor(pos.y);
112
113 //Get other corners
114 corners[1] = corners[0]; corners[1].x++;
115 corners[2] = corners[0]; corners[2].y++;
116 corners[3] = corners[2]; corners[3].x++;
117
118 //Get displacements
119 glm::vec2 displace[4];
120 displace[0] = pos - corners[0]; displace[1] = pos - corners[1];
121 displace[2] = pos - corners[2]; displace[3] = pos - corners[3];
122
123 //Get the dot products
124 float dots[4];
125
126 //Set dot products and check to ensure validity
127 for (int i = 0; i < 4; i++)
128 {
129 if (pos == corners[i])
130 {
131 dots[i] = 0.0f;
132 }
133 else if (corners[i].x < 0 || corners[i].x >= m_Resolution || corners[i].y < 0 || corners[i].y >= m_Resolution)
134 {
135 dots[i] = 0.0f;
136 }
137 else
138 {
139 if (m_ShareRandom)
140 {
141 dots[i] = glm::dot(displace[i], m_NoiseField[(int)corners[i].y].line[(int)corners[i].x].data);
142 }
143 else
144 {
145 int yIndex = (data * m_Resolution) + (int)corners[i].y;
146 dots[i] = glm::dot(displace[i], m_NoiseField[yIndex].line[(int)corners[i].x].data);
147 }
148 }
149 }
150
151 //Lerp dot product for point from the corners
152 float lX1 = std::lerp(dots[0], dots[1], pos.x - corners[0].x);
153 float lX2 = std::lerp(dots[2], dots[3], pos.x - corners[0].x);
154
155 float value = std::lerp(lX1, lX2, pos.y - corners[0].y);
156 value = glm::smoothstep(-1.0f, 1.0f, value);
157
158 int rangeIndex = data * 2;
159 float fullRange = m_Range[rangeIndex+1] - m_Range[rangeIndex];
160 value = ((value + 1.0f) * (fullRange/2.0f)) + m_Range[rangeIndex];
161
162 m_Data[y].line[x].data[data] = (uint8_t)value;
163 }
164 }
