130//Rotation
131template<typename T>
132void RotateShape(void* verticesArray, Component3f rotationCentre, float angle, Shape type, Axis axis)
133{
134 T* vertexPointer = (T*)verticesArray;
135
136 //Set number of vertices to translate
137 unsigned short int numberOfVertices = GetVerticesCount(type);
138
139 glm::vec3 axisVector = { 1.0f, 0.0f, 0.0f };
140 switch (axis)
141 {
142 case Axis::X:
143 axisVector = { 1.0f, 0.0f, 0.0f };
144 break;
145 case Axis::Y:
146 axisVector = { 0.0f, 1.0f, 0.0f };
147 break;
148 case Axis::Z:
149 axisVector = { 0.0f, 0.0f, 1.0f };
150 break;
151 }
152
153 //Translate for each vertice
154 for (int i = 0; i < numberOfVertices; i++) {
155 Vertex* vertex = (Vertex*)(void*)&vertexPointer[i];
156 glm::vec3 position = { vertex->position.a - rotationCentre.a, vertex->position.b - rotationCentre.b, vertex->position.c - rotationCentre.c };
157 position = glm::rotate(position, glm::radians(angle), axisVector);
158 vertex->position = { position.x + rotationCentre.a, position.y + rotationCentre.b, position.z + rotationCentre.c };
159 }
160}
